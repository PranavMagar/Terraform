'''Practical - 5[1]'''

import nltk
nltk.download('wordnet')
from nltk.corpus import wordnet as wn

# Display synsets
print(wn.synsets('happy'))                      # all synsets of 'happy'
print(wn.synsets('dog', pos=wn.VERB))           # synsets of 'dog' as verb

# Display name, definition, example of synsets
synsets = wn.synsets('car')
print("Synsets of car:", synsets)

for synset in synsets:
    print("\nSynset Name:", synset.name())
    print("Definition:", synset.definition())
    print("Examples:", synset.examples())

# Hypernym & hyponym
car = wn.synset("car.n.01")
print("\nCar Synset:", car)
print("Hypernyms of car:", car.hypernyms())
print("Hyponyms of car:", car.hyponyms())

# Meronym & Holonym
hand_syn = wn.synset('hand.n.01')
print("\nSynset for hand:", hand_syn)
print("Part Meronyms of hand:", hand_syn.part_meronyms())
print("Part Holonyms of hand:", hand_syn.part_holonyms())
